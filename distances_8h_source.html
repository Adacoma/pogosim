231inline b2Vec2 torus_delta(b2Vec2 a, b2Vec2 b, b2Vec2 dom_min, float Lx, float Ly) {
232 float ax = a.x - dom_min.x, ay = a.y - dom_min.y;
233 float bx = b.x - dom_min.x, by = b.y - dom_min.y;
